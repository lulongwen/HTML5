// 同步 alert， for
// 多线程 concurrent.thread.js

// 1 Concurrent.Thread.create 创建一个线程
Concurrent.Thread.create(function(){
    $('#id').click(function(){
        alert(1)
    });

    // 下面是一段特别复杂的函数
    for(var i=0;i<10000000;i++){
        console.log(i)
    };
});

// 2 script 标签创建线程
<script type="text/x-script.multithreaded-js">
    $('#id').click(function(){
        alert(1)
    });

    // 下面是一段特别复杂的函数
    for(var i=0;i<10000000;i++){
        console.log(i)
    };
</script>
